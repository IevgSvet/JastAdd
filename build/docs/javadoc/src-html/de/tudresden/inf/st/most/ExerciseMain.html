022        /**
023         * loads an expression language model
024         * @param fileName either a json file with a .json file extension or a DSL file with an .exp extension.
025         * @return an abstract syntax tree
026         * @throws IOException if the file could not be loaded
027         * @throws Parser.Exception if the parser could not parse the DSL file
028         */
029        public static Root load(String fileName) throws IOException, Parser.Exception {
030                URL expUrl = ExerciseMain.class.getClassLoader().getResource(fileName);
031                File file = null;
032                if (expUrl != null) {
033                        file = new File(expUrl.getFile());
034                } else {
035                        file = new File(fileName);
036                }
037                if (file == null) {
038                        throw new FileNotFoundException("Could not load JSON file " + fileName);
039                }
040    
041                if (fileName.endsWith(".json")) {
042                        System.out.println("Loading JSON file '" + fileName + "'.");
043                        return ASTNodeDeserializer.read(file);
044                } else {
045                        System.out.println("Loading expression DSL file '" + fileName + "'.");
046                        FileReader reader = new FileReader(file);
047                        ExpressionScanner scanner = new ExpressionScanner(reader);
048                        ExpressionParser parser = new ExpressionParser();
049                        return (Root) parser.parse(scanner);
050                }
051        }
